Semantic HTML Layout

Semantic HTML introduces meaning to the web page rather than just presentation. It helps with SEO and accessibility.

<header>
  <h1>My Portfolio</h1>
  <nav>
    <ul>
      <li><a href="#about">About</a></li>
      <li><a href="#projects">Projects</a></li>
    </ul>
  </nav>
</header>

<main>
  <section id="about">
    <h2>About Me</h2>
    <article>
      <p>I'm a web developer passionate about creating accessible websites.</p>
    </article>
  </section>
  
  <section id="projects">
    <h2>Projects</h2>
    <article>
      <h3>E-commerce Site</h3>
      <p>Built with HTML, CSS, and JavaScript.</p>
    </article>
  </section>
</main>

<footer>
  <p>© 2025 My Portfolio</p>
  <address>
    Contact me at <a href="mailto:example@email.com">example@email.com</a>
  </address>
</footer>

Modern HTML Examples

Responsive Table

<div class="table-container">
  <table>
    <caption>Employee Details</caption>
    <thead>
      <tr>
        <th scope="col">Name</th>
        <th scope="col">Department</th>
        <th scope="col">Salary</th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td data-label="Name">John Doe</td>
        <td data-label="Department">Engineering</td>
        <td data-label="Salary">$85,000</td>
      </tr>
    </tbody>
  </table>
</div>

Modern Form

<form class="modern-form">
  <div class="input-group">
    <label for="email">Email Address</label>
    <input type="email" id="email" 
           placeholder="Enter your email" 
           required>
  </div>
  
  <div class="input-group">
    <label for="password">Password</label>
    <input type="password" id="password" 
           placeholder="Enter your password" 
           minlength="8" required>
  </div>
  
  <div class="input-group">
    <label for="message">Message</label>
    <textarea id="message" 
              placeholder="Your message..." 
              rows="4"></textarea>
  </div>
  
  <button type="submit" class="btn">Submit</button>
</form>

Responsive Images

<picture>
  <source media="(min-width: 1200px)" 
          srcset="large.jpg">
  <source media="(min-width: 768px)" 
          srcset="medium.jpg">
  <img src="small.jpg" 
       alt="Descriptive alt text" 
       class="responsive-img">
</picture>

<figure>
  <img src="image.jpg" alt="Artwork">
  <figcaption>Beautiful landscape artwork</figcaption>
</figure>
