Live Example: Typewriter Effect

// Typewriter Effect
const text = "Brutal JavaScript in Action...";
let i = 0;

function typeWriter() {
  if (i < text.length) {
    document.getElementById("output").innerHTML += text.charAt(i);
    i++;
    setTimeout(typeWriter, 100);
  }
}

window.onload = typeWriter;
